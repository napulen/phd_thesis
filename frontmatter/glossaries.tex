\newglossaryentry{augmentednet}{name=AugmentedNet,description={An
Automatic Roman Numeral Analysis model, originally
introduced in \textcite{napoleslopez2021augmentednet}, and
extended in this \thesisdiss{}. The main model described
throughout the dissertation}}

\newglossaryentry{augsix}{name=augmented
sixth,description={A family of chords in Western tonal music
characterized by an ``augmented sixth'' interval, which
comprises at least three types of chords: the Italian
($\rnIt$), French ($\rnFr$), and German ($\rnGer$) augmented
sixth chords}}

\newglossaryentry{closed-position}{name=closed-position
form,description={In the context of chord realizations, a
closed-position chord is an arrangement of the chord,
usually in four parts, such that the three upper voices span
a maximum interval of an octave}}

\newglossaryentry{commonpractice}{name=common-practice
period, description={In Western art music, this refers to
the period of music roughly between 1650--1900, which is
characterized by the use of the tonal system}}

\newglossaryentry{keyprofile}{name=key profile,
description={A key profile is a pitch-class distribution
used to correlate pitch histograms with a certain key or
scale. They were arguably introduced for the first time in
\textcite{krumhansl1979quantification,krumhansl1982tracing},
and applied on global-key estimation problems in
\textcite{krumhansl1990cognitive}}}

\newglossaryentry{lineoffifths}{name=line of
fifths,description={A line of fifths is an extension of the
more commonly known ``circle of fifths''. A line of fifths
occurs when enharmonic nonequivalence is assumed in the
circle of fifths. In the circle of fifths, the pitches in
the circle repeat because it is assumed that two enharmonic
pitches have the same location in the circle (e.g.,
$\pitchGb$ and $\pitchFs$). In the line of fifths, however,
the ``sharper'' and ``flatter'' pitches are, the further
away they are from the origin, and hence, from each other.
See \textcite{temperley2000line} for an extensive discussion
on the topic}}

\newglossaryentry{melisma}{name=Melisma,description={The
\emph{Melisma Music Analyzer (version 1)} is a suite of
programs for music analysis. One of the analytical features
of the suite is automatic Roman numeral analysis. After the
contributions in \textcite{sapp2009tsroot}, the
\emph{Melisma} Roman numeral analysis module probably became
the first end-to-end Roman numeral analysis system. See
\textcite{napoleslopez2017automatic} for a discussion of the
Roman numeral analysis component of \emph{Melisma}, and see
\textcite{temperley2004cognition} for the underlying
research behind the algorithms of the system}}

\newglossaryentry{musicxml}{name=MusicXML, description={A
symbolic music format based on \gls{xml}}}

\newglossaryentry{neapolitan}{name=Neapolitan,description={A
special type of chord in Western tonal music, analogous to a
$\musFlat\rn{II}$ Roman numeral}}

\newglossaryentry{nonchord}{name=nonchord
tone,description={A nonchord tone is an ornamental note that
is not a member of the tones of a given chord, but that
sounds in combination or proximity with them for aesthetic
reasons}}

\newglossaryentry{overfit}{name=overfit,
description={Overfitting is a phenomenon where a machine
learning model excessively fits to the training data,
compromising its capability to generalize on
new---distinct---data. It is generally considered an
undesirable effect when training a machine learning model}}

\newglossaryentry{pitchclass}{name=pitch
class,description={In Western music, a pitch class is each
of the 12 pitches that conform the \gls{12tet} chromatic
scale, regardless of their octave and/or pitch spelling}}

\newglossaryentry{realize}{name=realize, description={In the
context of chords, a chord realization is an arrangement of
a chord label into notes. For example, if a chord is
indicated as $\rnwkey{F}{\rnVsev}$, one realization consists
of the tuple of notes $(\pitchC{}4, \pitchE{}4, \pitchG{}4,
\pitchBb{}4)$, whereas a different realization consists of
the notes $(C4, G4, E5, B\flat5)$}}

\newglossaryentry{romantext}{name=RomanText, description={A
digital standard for Roman numeral analysis}}

\newglossaryentry{spine}{name=spine, description={In the
context of the Humdrum format, a spine is a column of data}}
